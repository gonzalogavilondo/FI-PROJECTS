Proceso Grupo_personas

	Definir edad, conteo_pmayor, i como enteros;
	conteo_pmayor <- 0;
	Para i<-1 Hasta 30 Hacer
		Escribir i,".Digite su edad: ";
		Leer edad;
		Si edad > 18 Entonces
			conteo_pmayor <- (conteo_pmayor + 1);
		FinSi
	FinPara
	Escribir "Las personas mayores a 18 años son: ", conteo_pmayor;

FinProceso
